<script>
let inventario = JSON.parse(localStorage.getItem("inventario")) || {};
let historial = JSON.parse(localStorage.getItem("historial")) || [];
actualizarTabla();
cargarHistorial();
function guardarDatos(){
    localStorage.setItem(
        "inventario",
        JSON.stringify(inventario)
    );
    localStorage.setItem(
        "historial",
        JSON.stringify(historial)
    );
}
function actualizarTabla(){
    let tabla = document.getElementById("tablaInventario");
    tabla.innerHTML = "";
    for(let producto in inventario){
        let cantidad = inventario[producto];
        let estado = "";
        let clase = "";
        if(cantidad <= 5){
            estado = "Bajo";
            clase = "rojo";
        }
        else if(cantidad <= 15){
            estado = "Medio";
            clase = "amarillo";
        }
        else{
            estado = "Alto";
            clase = "verde";
        }
        tabla.innerHTML += `
        <tr>
            <td>${producto}</td>
            <td>${cantidad}</td>
            <td>
                <span class="${clase}">
                    ${estado}
                </span>
            </td>
        </tr>
        `;
    }
    guardarDatos();
}
function agregarEntrada(){
    let producto = document.getElementById("producto").value;
    let cantidad = parseInt(
        document.getElementById("cantidad").value
    );
    if(producto === "" || isNaN(cantidad)){
        alert("Completa los campos");
        return;
    }
    if(inventario[producto]){
        inventario[producto] += cantidad;
    }
    else{
        inventario[producto] = cantidad;
    }
    registrarHistorial(
        producto,
        "Entrada",
        cantidad
    );
    actualizarTabla();
    limpiarCampos();
}
function registrarSalida(){
    let producto = document.getElementById("producto").value;
    let cantidad = parseInt(
        document.getElementById("cantidad").value
    );
    if(producto === "" || isNaN(cantidad)){
        alert("Completa los campos");
        return;
    }
    if(!inventario[producto]){
        alert("Producto no existe");
        return;
    }
    if(inventario[producto] < cantidad){
        alert("No hay suficiente inventario");
        return;
    }
    inventario[producto] -= cantidad;
    registrarHistorial(
        producto,
        "Salida",
        cantidad
    );
    actualizarTabla();
    limpiarCampos();
}
function registrarHistorial(
    producto,
    movimiento,
    cantidad
){
    let fechaActual = new Date();
    let fecha =
        fechaActual.toLocaleDateString();
    let hora =
        fechaActual.toLocaleTimeString();
    historial.push({
        producto,
        movimiento,
        cantidad,
        fecha,
        hora
    });
    cargarHistorial();
    guardarDatos();
}
function cargarHistorial(){
    let tabla =
        document.getElementById(
            "tablaHistorial"
        );
    tabla.innerHTML = "";
    historial.forEach(item => {
        tabla.innerHTML += `
        <tr>
            <td>${item.producto}</td>
            <td>${item.movimiento}</td>
            <td>${item.cantidad}</td>
            <td>${item.fecha}</td>
            <td>${item.hora}</td>
        </tr>
        `;
    });
}
function limpiarCampos(){
    document.getElementById(
        "producto"
    ).value = "";
    document.getElementById(
        "cantidad"
    ).value = "";
}
function mostrarInventario(){
    document.getElementById(
        "inventario"
    ).style.display = "block";
    document.getElementById(
        "historial"
    ).style.display = "none";
}
function mostrarHistorial(){
    document.getElementById(
        "inventario"
    ).style.display = "none";
    document.getElementById(
        "historial"
    ).style.display = "block";
}
function descargarExcel(){
    let contenido =
    "Producto,Movimiento,Cantidad,Fecha,Hora\n";
    historial.forEach(item => {
        contenido +=
        `${item.producto},${item.movimiento},${item.cantidad},${item.fecha},${item.hora}\n`;
    });
    let blob = new Blob(
        [contenido],
        {type:"text/csv"}
    );
    let enlace =
        document.createElement("a");
    enlace.href =
        URL.createObjectURL(blob);
    enlace.download =
        "inventario_la_terraza.csv";
    enlace.click();
}
